Build a DFT test harness that validates the calculation engine against reference data from the database, covering all 5 products across multiple section types, fire ratings, and failure temps.

Purpose: This is the safety net that ensures the DFT lookup chain returns correct values. Since this is a fire safety specification tool, every DFT value must be deterministically correct -- not approximate. The test harness runs on every build and catches any regression.
Output: A pytest test suite with reference data fixture that validates DFT lookups across all products and reports specific mismatches.

**tests/test_dft_lookup.py:**

```python
import json
import os
import pytest
from app.services import dft_service

FIXTURE_PATH = os.path.join(os.path.dirname(__file__), 'fixtures', 'dft_reference_data.json')

def load_reference_data():
    with open(FIXTURE_PATH) as f:
        return json.load(f)

# Parametrized test: run every reference case
class TestDFTReferenceData:
    """Validate DFT lookups against reference data extracted from database."""
    
    @pytest.fixture(autouse=True)
    def setup(self, app):
        self.app = app
    
    def reference_cases(self):
        data = load_reference_data()
        return [(d, d.get('description', f"{d['section_name']}_{d['hp_profile_name']}_{d['product_name']}")) 
                for d in data if d['expected_status'] == 'ok']
    
    def test_dft_reference_data(self, app):
        """Run all reference cases and report mismatches."""
        data = load_reference_data()
        ok_cases = [d for d in data if d['expected_status'] == 'ok']
        
        failures = []
        with app.app_context():
            for case in ok_cases:
                result = dft_service.lookup_dft(
                    case['section_id'],
                    case['hp_profile_name'],
                    case['product_id'],
                    case['fire_rating_id'],
                    case['failure_temp_id']
                )
                
                if result['status'] != 'ok':
                    failures.append(f"LOOKUP FAILED: {case['section_name']} {case['hp_profile_name']} "
                                  f"{case['product_name']} {case['fire_rating']} {case['failure_temp']} "
                                  f"-> status={result['status']}, error={result.get('error')}")
                    continue
                
                if abs(result['dft_mm'] - case['expected_dft_mm']) > 0.001:
                    failures.append(f"DFT MISMATCH: {case['section_name']} {case['hp_profile_name']} "
                                  f"{case['product_name']} {case['fire_rating']} {case['failure_temp']} "
                                  f"-> got {result['dft_mm']}, expected {case['expected_dft_mm']}")
                
                if result['hp_over_a'] != case['expected_hp_over_a']:
                    failures.append(f"HP/A MISMATCH: {case['section_name']} {case['hp_profile_name']} "
                                  f"-> got {result['hp_over_a']}, expected {case['expected_hp_over_a']}")
        
        if failures:
            report = f"\n{len(failures)} failures out of {len(ok_cases)} cases:\n"
            report += "\n".join(f"  - {f}" for f in failures[:20])  # Show first 20
            if len(failures) > 20:
                report += f"\n  ... and {len(failures) - 20} more"
            pytest.fail(report)

class TestDFTKnownValues:
    """Hard-coded tests for verified DFT values."""
    
    def test_sc802_uc4_254x254x73_60min_550c(self, app):
        """Verified reference point from database analysis."""
        with app.app_context():
            r = dft_service.lookup_dft(750, 'U4', 266, 3, 7)
            assert r['status'] == 'ok'
            assert r['hp_over_a'] == 155.0
            assert r['coverage_id'] == 720
            assert r['dft_mm'] == 0.367
            assert r['use_band'] == 'C'
    
    def test_sc601_varying_use_band(self, app):
        """SC601 U4 uses different coverage bands at different fire ratings."""
        with app.app_context():
            r_30min = dft_service.lookup_dft(750, 'U4', 278, 1, 4)  # 30min
            r_60min = dft_service.lookup_dft(750, 'U4', 278, 3, 7)  # 60min
            
            # Both should succeed (or at least one should if temps are valid)
            # The key assertion: if both succeed, they should use different use_bands
            if r_30min['status'] == 'ok' and r_60min['status'] == 'ok':
                # SC601 U4 at 30min uses use_band='B', at 60min uses use_band='C'
                assert r_30min['use_band'] != r_60min['use_band'], \
                    f"Expected different use_bands, both got '{r_30min['use_band']}'"

class TestDFTInvalidInputs:
    """Error handling for invalid inputs."""
    
    def test_invalid_section(self, app):
        with app.app_context():
            r = dft_service.lookup_dft(999999, 'U4', 266, 3, 7)
            assert r['status'] != 'ok'
            assert r['dft_mm'] is None
    
    def test_invalid_profile(self, app):
        with app.app_context():
            r = dft_service.lookup_dft(750, 'NONEXISTENT', 266, 3, 7)
            assert r['status'] != 'ok'
            assert r['dft_mm'] is None
    
    def test_invalid_product(self, app):
        with app.app_context():
            r = dft_service.lookup_dft(750, 'U4', 999999, 3, 7)
            assert r['status'] != 'ok'
            assert r['dft_mm'] is None

class TestDFTProductCoverage:
    """Every product must have at least one valid DFT lookup."""
    
    @pytest.mark.parametrize("product_id,product_name", [
        (266, "SC802"), (278, "SC601"), (283, "SC801-120"), 
        (284, "SC901"), (324, "SC804")
    ])
    def test_product_has_valid_dft(self, app, product_id, product_name):
        with app.app_context():
            # Try a common section+profile combo
            r = dft_service.lookup_dft(750, 'U4', product_id, 3, 7)
            if r['status'] != 'ok':
                # Try 550C (failure_temp_id=7) and 500C (6) and 400C (4)
                for ft in [6, 5, 4, 8, 9, 10]:
                    r = dft_service.lookup_dft(750, 'U4', product_id, 3, ft)
                    if r['status'] == 'ok':
                        break
            assert r['status'] == 'ok', \
                f"Product {product_name} ({product_id}) has no DFT for UC 254x254x73 at 60min"

class TestDFTApiEndpoint:
    """Integration test via HTTP."""
    
    def test_post_lookup(self, client):
        r = client.post('/api/dft/lookup', json={
            'section_id': 750,
            'hp_profile_name': 'U4',
            'product_id': 266,
            'fire_rating_id': 3,
            'failure_temp_id': 7
        })
        assert r.status_code == 200
        assert r.json['status'] == 'ok'
        assert r.json['dft_mm'] == 0.367
    
    def test_post_missing_fields(self, client):
        r = client.post('/api/dft/lookup', json={'section_id': 750})
        assert r.status_code == 400
```
